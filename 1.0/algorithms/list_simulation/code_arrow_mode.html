class Node:
    Node *prev
    Node *next
    key

class LinkedList:
    Node *nil  # 哨兵

    init():
        nil ← 產生節點
        nil->next ← nil  # 指向自己
        nil->prev ← nil  # 指向自己

    insert(data):
        # 生成節點並設定其資料與指標
        Node *x ← ノードを生成
        x->key ← data # 設定資料內容
        x->next ← nil->next
        x->prev ← nil
        # 設定哨兵及原本位於最前端的節點指標
        nil->next->prev ← x
        nil->next ← x

    listSearch(k):
        Node *cur ← nil->next # 從哨兵正後方的元素開始追蹤
        while cur ≠ nil and cur->key ≠ k:
            cur ← cur->next
        return cur

    deleteNode(Node *t):
        if t = nil: return # 若 t 為哨兵則不進行處理
        t->prev->next ← t->next
        t->next->prev ← t->prev
        釋放 t 的記憶體

    deleteKey(k):
        # 將取得的節點刪除
        deleteNode(listSearch())

#虛擬碼
class Queue
    Q          # 儲存佇列元素的陣列
    head ←0      
    tail ←0      

    init():
        head ← 0   #初始化head及tail
        tail ← 0
  
    enqueue(x):
        Q[tail++] ← x       # 指定x之後,將tail加1

    dequeue():
        return Q[head++]         # 傳回Q[head]的值後,將head加1

    empty():
        return head = tail       # 當head等於tail時,傳回真

#模擬演算法動畫中的佇列操作
Queue que(6)
que.enqueue(4)
que.enqueue(8)
que.enqueue(1)
que.dequeue()
que.enqueue(7)
que.dequeue()

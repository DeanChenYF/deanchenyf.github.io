class Node:
    Node *prev
    Node *next
    key

class LinkedList:
    Node *nil  # 番兵

    init():
        nil ← ノードを生成
        nil->next ← nil  # 自分自身を指す
        nil->prev ← nil  # 自分自身を指す

    insert(data):
        # ノードを生成してデータとポインタを設定
        Node *x ← ノードを生成
        x->key ← data # データ本体を設定
        x->next ← nil->next
        x->prev ← nil
        # 番兵と、元の先頭ノードのポインタを設定
        nil->next->prev ← x
        nil->next ← x

    listSearch(k):
        Node *cur ← nil->next # 番兵の次の要素から辿る
        while cur ≠ nil and cur->key ≠ k:
            cur ← cur->next
        return cur

    deleteNode(Node *t):
        if t = nil: return # t が番兵の場合は処理しない
        t->prev->next ← t->next
        t->next->prev ← t->prev
        tのメモリを解放

    deleteKey(k):
        # 検索したノードを削除
        deleteNode(listSearch())

#虛擬碼

class Stack:
    S       # 管理元素的陣列
    top     # 指向堆疊頂端的箭頭

    init():
        top ← -1     # 初始化堆疊

    push(x):
        S[++top] ← x # 先將top加１後指向的元素指定為X

    pop():
        return S[top--]   # 傳回S[top]後,再將top減１

    peak()
        return S[top]     # 傳回S[top]

    empty():
        return top = -1   # 若top為-1,表示為空堆疊

    size():
        return top + 1    # 若top加1


# 模擬演算法動畫中的堆疊操作

Stack st(5) 
st.push(8)
st.push(6)
st.push(7)
st.pop()
